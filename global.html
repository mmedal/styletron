connectToStyles(styleFn) → {function}

Source:
  • styletron-react/src/connect-to-styles.js, line 35

Higher-order component for presentational components

Example
class Panel extends React.Component {
  render() {
    <div className={this.styles.container}>
      <h1 className{this.styles.title}>Panel</h1>
      {this.props.children}
    </div>
  }
}

const hoc = connectToStyles(props => {
  return {
    container: {
      width: '640px',
      background: props.color
    },
    title: {
      fontSize: '40px'
    }
  };
});

module.exports = hoc(Panel);
